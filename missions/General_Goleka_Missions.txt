					random_trade_node_member_province = {
						limit = {
							owned_by = ROOT

							is_capital = no
							development = 20
						}

					}

							owned_by = ROOT

							owned_by = ROOT
							development = CAPITAL

					}
				}
			}
			if = {
				limit = { has_saved_event_target = reward_province }
